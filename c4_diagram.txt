workspace "Brazo Robotico" "Control de brazo robótico con gemelo digital" {
    model {
        user = person "Estudiante/Ingeniero" "Opera el brazo robótico y visualiza la telemetría y simulación 3D a través de la interfaz." {
            tags "User"
        }
        camera = element "Cámara" "Camera" "Cámara web conectada mediante USB"
        robot = element "OpenBotV v1" "Robot" "Microcontrolador OpenCM 9.04 conectado mediante USB"
        softwareSystem = softwareSystem "Sistema Brazo Robotico" "Sistema integral de control, visión artificial y simulación física." {
            tags "App"

            # ============================================================
            #  C2 - CONTENEDORES
            # ============================================================
            mainWindow = container "Main Window (App Orchestrator)" "Ensambla mixins de UI, crea el DataController y los controladores de features y servicios." "Python/PyQt6"

            uiFeatures = container "Features (Capa de Funcionalidades)" "Módulos funcionales (patrón Controller/Widget/Worker) que capturan la intención del usuario y publican en el bus de señales." "Python/PyQt6" {
                kinematicsUI = component "Kinematics Feature" "Control por coordenadas cartesianas usando cinemática inversa. Resuelve IK a petición del flujo Pick and Place."
                slidersUI = component "Sliders Feature" "Control manual de articulaciones mediante deslizadores."
                simulationUI = component "Simulation Feature" "Visualización 3D del gemelo digital; integra Quick3D con el motor de física." "QML/Quick3D"
                cameraUI = component "Camera Feature" "Gestiona el flujo de video; el controlador es el único puente del worker hacia el bus."
                graphUI = component "Graph Feature" "Representación gráfica de telemetría en tiempo real."
                pickAndPlaceUI = component "Pick And Place Feature" "Secuencia de selección y colocación de objetos mediante máquina de estados."
                calibrationUI = component "Calibration/Color Window" "Ventanas independientes de calibración de cámara y detección de color."
                tags "Features"
            }

            dataMgmt = container "Data Management (Sync Manager)" "Núcleo de sincronización. Único orquestador cross-feature; normaliza y despacha datos entre la UI y los servicios de ejecución." "Python/PyQt6" {
                dataController = component "DataController" "Mediador reactivo (Cerebro). Escucha los buses de las features, aplica la lógica del gemelo digital y comanda los servicios." {
                    tags "Data"
                }
                signalManagers = component "Signal Managers" "Bus reactivo Pub/Sub con un gestor por dominio: Simulation, Physical, Camera, Search, DrawView, PickPlace, Sliders, Kinematics, Theme y Config."
                configManager = component "ConfigManager" "Persistencia de configuración (settings, calibraciones de cámara) en archivos JSON." {
                    tags "configManager"
                }
                timers = component "Timers" "GlobalTimer (tick de sincronización sim/robot) y FrameCounter (cadencia de visión)."
                tags "Core"
            }

            backServices = container "Services (Capa de Ejecución)" "Motores de ejecución sin lógica de orquestación; reciben comandos y devuelven telemetría." "Python" {
                robotService = component "Robot Service" "Comunicación serial bidireccional con el OpenCM 9.04 (envío de PWM, lectura de telemetría)." "Python/Serial"
                simulationService = component "Simulation Service" "Motor de física PyBullet y sincronización del gemelo digital." "Python/PyBullet"
                visionService = component "Vision Service" "Algoritmos de visión: detección ChArUco, segmentación de esferas y estimación de pose 3D." "Python/OpenCV"
                deviceService = component "Device Service" "Monitoreo plug-and-play de cámaras USB y puertos serie."
                stylingUiService = component "Styling/UI Service" "Temas claro/oscuro y componentes visuales de realimentación (toasts, diálogos)."
                tags "Services"
            }

            # ============================================================
            #  RELACIONES
            # ============================================================

            # --- Contexto (C1) ---
            user -> mainWindow "Usa la aplicación"
            camera -> cameraUI "Proporciona el video una vez seleccionada la cámara"
            robot -> robotService "Lectura/escritura de ángulos de motores y temperaturas"

            # --- Ensamblaje (Main Window) ---
            mainWindow -> uiFeatures "Inicializa y ensambla los controladores"
            mainWindow -> dataController "Crea el orquestador central"
            mainWindow -> robotService "Inicializa la conexión serial"

            # --- Flujo de Comandos (Features -> DataController, vía bus propio) ---
            slidersUI -> dataController "Publica ángulos y cambio de modo (SlidersSignalManager)"
            kinematicsUI -> dataController "Publica objetivos cartesianos y cambio de modo (KinematicsSignalManager)"
            pickAndPlaceUI -> dataController "Solicita movimientos, IK y control de búsqueda (PickPlaceSignalManager)"
            cameraUI -> dataController "Publica poses y detecciones; solicita limpiar esferas (CameraSignalManager)"

            # --- Orquestación cross-feature mediada por el DataController ---
            dataController -> kinematicsUI "Rutea solicitudes de cinemática inversa"
            dataController -> pickAndPlaceUI "Devuelve resultados de IK y detecciones de esferas"
            dataController -> simulationUI "Actualiza el estado visual 3D"
            dataController -> graphUI "Envía datos para visualización de gráficos"

            # --- Infraestructura interna del Sync Manager ---
            dataController -> signalManagers "Se suscribe y despacha sobre los buses de dominio"
            dataController -> configManager "Lectura/escritura de configuración y calibraciones"
            dataController -> timers "Recibe ticks de sincronización"

            # --- DataController <-> Servicios de Ejecución ---
            dataController -> robotService "Comanda el brazo físico (Modo Físico)"
            dataController -> simulationService "Sincroniza el gemelo digital (Modo Simulación)"
            robotService -> dataController "Retorna telemetría del hardware"
            simulationService -> dataController "Retorna telemetría de simulación"

            # --- Flujo de Visión (el Camera Feature posee el worker) ---
            cameraUI -> visionService "Envía frames para procesamiento"
            visionService -> cameraUI "Devuelve ChArUco, esferas y poses 3D detectadas"

            # --- Simulación 3D ---
            simulationUI -> simulationService "Usa el motor de física PyBullet"

            # --- Dispositivos (plug-and-play) ---
            camera -> deviceService "Configura y actualiza lista de dispositivos de video"
            robot -> deviceService "Configura y actualiza lista de puertos COM"
            deviceService -> cameraUI "Reporta cámaras disponibles"
            deviceService -> robotService "Reporta puertos seriales activos"

            # --- Estilo / UI transversal ---
            stylingUiService -> uiFeatures "Provee temas y componentes de realimentación"
        }
    }
    views {
        # ---- C1: Contexto ----
        systemContext softwareSystem "SystemContext" {
            include *
            description "Contexto del sistema: usuario, cámara USB y robot físico."
            # autoLayout
        }

        # ---- C2: Contenedores ----
        container softwareSystem "Containers" {
            include *
            description "Separación en cuatro capas: Main Window, Features, Data Management (Sync Manager) y Services."
            # autoLayout
        }

        # ---- C3: Componentes (uno por contenedor) ----
        component uiFeatures "UIFeaturesComponents" {
            include *
            description "Features de usuario. Cada una sigue el patrón Controller/Widget/Worker y publica en su bus de dominio."
            # autoLayout
        }
        component dataMgmt "DataManagementComponents" {
            include *
            description "Núcleo de sincronización: el DataController media sobre los buses, la configuración y los temporizadores."
            # autoLayout
        }
        component backServices "ServicesComponents" {
            include *
            description "Motores de ejecución: robot, simulación física, visión, dispositivos y estilo."
            # autoLayout
        }

        styles {
            element "Element" {
                shape roundedBox
                background #F5F5F5
            }
            element "App" {
                shape Window
            }
            element "User" {
                metadata true
                shape person
            }
            element "Features" {
                background #1168bd
                color #F5F5F5
            }
            element "configManager" {
                shape folder
                background #FFD561
                color #4D4D4D
            }
            element "Data" {
                background #cf1c1c
                color #F5F5F5
            }
            element "Core" {
                background #8a1c8a
                color #F5F5F5
            }
            element "Services" {
                background #207a2c
                color #F5F5F5
            }
        }
    }
}
